namespace = auto_council


auto_council.0001 = {
	type = character_event
	title = auto_council_0001_t
	desc = auto_council_0001_desc
	theme = faith

	left_portrait = {
		character = root
		animation = personality_rational
	} 
	right_portrait = {
		character = cp:councillor_court_chaplain
		animation = personality_zealous
	}

	immediate = {
		debug_log = auto_council_debug_0007	#Firing event...
	}
		
	option = { # Start the Conversion
		name = auto_council_0001_a
		trigger = {
			any_sub_realm_county = {
				NOT = {
					faith = root.faith
				}
			}
			exists = cp:councillor_court_chaplain
		}			

		trigger_event = auto_council.0002
	}
	
	option = { # Cancel
		name = auto_council_0001_b			
	}
}

auto_council.0002 = {
	hidden = yes
	type = character_event

	immediate = {
		debug_log = auto_council_debug_0003 #Firing event...
		
		every_sub_realm_county = {
			if = {
				limit = { NOT = { faith = root.faith }}
				add_to_list = heathens
			}
		}

		if = {
			limit = {
				any_in_list = {
					list = heathens
				}
				exists = cp:councillor_court_chaplain			
			}			

			count_items_in_list_effect = {
				LCOUNT = heathens_count
				LVAR = heathens
			}			
						
			debug_log = auto_council_debug_0001 #there is X counties in list...
				
			send_interface_message = {
				type = event_chancellor_task_good
				right_icon = cp:councillor_court_chaplain
				title = auto_council_0002_mes1_t
				desc = auto_council_0002_mes1_description
				tooltip = auto_council_0002_mes1_description					
			}

			random_in_list = {
				list = heathens
				save_scope_as = selected_county
				remove_from_list = heathens
			}

			count_items_in_list_effect = {
				LCOUNT = heathens_count
				LVAR = heathens
			}	
			
			set_variable = {
				name = saved_scope
				value = scope:selected_county
			}				

			if = {
				limit = { 
					exists = scope:selected_county
					cp:councillor_court_chaplain.liege = root
				}

				debug_log = auto_council_debug_0005 # all evaluated to true...
			
				debug_log = auto_council_debug_0002 #starting conversion of...

				cp:councillor_court_chaplain = {				
					set_council_task = {
						task_type = task_conversion
						target = scope:selected_county.title_province
					}				
				}							
			}			

			else = {				
				debug_log = auto_council_debug_0004 # at least one evaluated to false															
			}

			#remove_variable = heathens_count
			#remove_variable = saved_scope
		}	
	}
}
