namespace = auto_council

#
# Events for decisions
# Range - 0900-0999
#

auto_council.0900 = {
	type = character_event
	title = auto_council_0001_t
	desc = auto_council_0001_desc
	theme = faith

	left_portrait = {
		character = root
		animation = personality_rational
	} 
	right_portrait = {
		character = cp:councillor_court_chaplain
		animation = personality_zealous
	}

	immediate = {
		debug_log = auto_council_debug_0001	#Firing event...
	}
		
	option = { # Start the Conversion
		name = auto_council_0001_a
		trigger = {
			any_sub_realm_county = {
				NOT = {
					faith = root.faith
				}
			}
			exists = cp:councillor_court_chaplain
		}
		if = { limit = { NOT = { has_character_flag = allow_automatic_convert_religion} }
			add_character_flag = { flag = allow_automatic_convert_religion}
		}			
		

		trigger_event = auto_council.0002
	}
	
	option = { # Cancel
		name = auto_council_0001_b			
	}
}

auto_council.0901 = {
	type = character_event
	title = auto_council_0003_t
	desc = auto_council_0003_desc
	theme = culture_change

	left_portrait = {
		character = root
		animation = personality_rational
	} 
	right_portrait = {
		character = cp:councillor_steward
		animation = personality_honorable
	}

	immediate = {
		debug_log = auto_council_debug_0007	#Firing event...
	}

	option = { # Start the Conversion
		name = auto_council_0003_a
		trigger = {
			any_sub_realm_county = {
				NOT = {
					culture = root.culture
				}
			}
			exists = cp:councillor_steward
		}	
		if = { limit = { NOT = { has_character_flag = allow_automatic_promote_culture} }		
			add_character_flag = { flag = allow_automatic_promote_culture }
		}

		trigger_event = auto_council.0004
	}	

	option = { # Cancel
		name = auto_council_0003_b			
	}
}
